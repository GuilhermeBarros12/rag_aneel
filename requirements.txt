# ── Versão mínima do Python ─────────────────────────────────────
# Este projeto requer Python 3.10 ou superior.
# Verifique sua versão com: python --version

# ── Extração e chunking de documentos ──────────────────────────
# langchain-text-splitters: fornece o RecursiveCharacterTextSplitter usado no chunking
langchain-text-splitters
# PyMuPDF: engine de leitura de PDFs (dependência do pymupdf4llm)
PyMuPDF
# pymupdf4llm: converte PDF → Markdown preservando tabelas e formatação
pymupdf4llm
# beautifulsoup4: extrai texto limpo de arquivos HTML
beautifulsoup4
# lxml: parser HTML rápido, usado internamente pelo BeautifulSoup
lxml

# ── Download e automação web ────────────────────────────────────
# selenium: automação do Chrome para downloads de PDF com redirect
selenium
# webdriver-manager: instala e gerencia o ChromeDriver automaticamente
webdriver-manager
# requests: download HTTP direto para arquivos .html e .xlsx
requests

# ── Embeddings e busca vetorial ─────────────────────────────────
# sentence-transformers: modelo paraphrase-multilingual-mpnet-base-v2 (embeddings em PT)
sentence-transformers
# chromadb: banco vetorial com persistência em disco, sem servidor
chromadb

# ── LLM (geração de respostas) ──────────────────────────────────
# google-genai: cliente oficial da API Gemini (novo SDK)
google-genai
# google-generativeai: cliente oficial da API Gemini (SDK legado usado no avaliar.py)
google-generativeai
# groq: cliente oficial da Groq API — fallback gratuito
groq

datasets

# tqdm: barra de progresso nos loops de download e indexação
tqdm
# python-dotenv: carrega variáveis do .env sem expor no código
python-dotenv
# huggingface_hub: download automático do vectorstore pré-construído
huggingface_hub
